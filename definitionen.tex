\else % we are running PDFLaTeX
  \ifnum \pdfoutput=0
    \pdffalse % HTML- bzw. Postscript
    #1
  \else
    \ifnum\pdfoutput=1 % PDF
    \pdftrue
    #2
    \fi
  \fi
\fi
}
